9            Scanner FileReader = new Scanner(System.in);

11           boolean ValidFile = false;

13               try {
14                   Reader = new Scanner(new File(FileReader.next()));

16               } catch (FileNotFoundException e) {
17                   System.out.println("Please enter a valid file name. Make sure it's in the working directory.");
18                   System.out.println(e.getLocalizedMessage());

24       }
25       //takes the scanner, reads the file and puts it into a string, and then feeds the string to the enigma machine
26       private static String PreEncrypt(Scanner ReadFile) {
27           StringBuilder PlainText = new StringBuilder();
28           while (ReadFile.hasNext()) {
29               PlainText.append(" ").append(ReadFile.next());
30           }
31           return PlainText.toString();
32       }
